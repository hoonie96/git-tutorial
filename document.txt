create a remote repository from www.github.com
find local repository and clone the url of remote repository [git clone "url"]
create a file in local repostiory
// Initialise local git repository
git init
// Add files to index
git add <file>
// Check status of working tree
git status
// Commit changes in index
git commit -m "note"
// Push to remote repository
git push
// Pull latest from remote repository
git pull
// Clone repository into a new directory
git clone

// Git global setup
git config --global user.name "name"
git config --global user.email "email"

// Process
1working directory
->git add
2staging area
->git commit
3local repository
->git push
4remote repository
<-git fetch
3local repository
<-git merge
1working directory
(git pull = git fetch + git merge)

// To undo
git reset <file>
// To restore
git checkout -- <file>
git restore <file>
// To rewrite commit message
git commit --amend
// To unstage
git rm --cached <file>
